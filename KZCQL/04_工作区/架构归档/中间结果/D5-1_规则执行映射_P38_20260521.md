### 4.3 建议的LEN-01检查项设计

```bash
# 建议新增check_len01()函数
check_len01() {
    local md_file="$1"
    local log_file="$2"
    
    # 统计段落信息
    local total_paragraphs=$(grep -c '^[[:alnum:]]' "$md_file")
    local long_paragraphs=$(awk '/^[[:alnum:]]/{if(length>80)count++}END{print count}' "$md_file")
    local ratio=$(echo "scale=2; $long_paragraphs / $total_paragraphs * 100" | bc)
    
    # 判断标准: >80字段落≤10%
    if (( $(echo "$ratio <= 10" | bc -l) )); then
        echo "  ✓ LEN-01 段落长度检查通过 ($long_paragraphs/$total_paragraphs=$ratio%)" >> "$log_file"
        return 0
    else
        echo "  ✗ LEN-01 段落长度检查失败 ($long_paragraphs/$total_paragraphs=$ratio% > 10%)" >> "$log_file"
        return 1
    fi
}
```

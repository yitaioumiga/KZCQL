```python
# 在AgentCallRecord中增加parallel标记
class AgentCallRecord:
    def __init__(self, agent_name: str, call_time: str, status: str, 
                 output_path: str = None, is_parallel: bool = False):
        self.agent_name = agent_name
        self.call_time = call_time
        self.status = status
        self.output_path = output_path
        self.is_parallel = is_parallel

# 新增并行调用方法
def call_parallel_agents(self, agent_names: List[str], 
                         agent_funcs: Dict[str, Callable]) -> Dict:
    """
    并行调用多个子智能体
    
    适用于：D5四个维度（A1-D5-1/2/3/4）
    """
    from concurrent.futures import ThreadPoolExecutor
    
    results = {}
    with ThreadPoolExecutor(max_workers=len(agent_names)) as executor:
        futures = {
            executor.submit(agent_funcs[name], {}): name 
            for name in agent_names
        }
        for future in futures:
            name = futures[future]
            try:
                results[name] = future.result()
            except Exception as e:
                results[name] = {"status": "failed", "error": str(e)}
    
    return results
```

**验证方法**：检查orchestrator.py是否支持并行调用
